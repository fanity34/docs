---
title: "Application Details"
api: "GET https://api.centralcart.com.br/api/application"
authMethod: "bearer"
description: "Get information about your store."
---

### Response

<ResponseField name="id" type="number">
  The ID of your store
</ResponseField>

<ResponseField name="name" type="string">
  The name of your store
</ResponseField>

<ResponseField name="plan" type="string">
  The plan of your store
</ResponseField>

<ResponseField name="overdue_date" type="string">
  The overdue date of your store
</ResponseField>

<ResponseField name="url" type="string">
  The url of your store
</ResponseField>

<ResponseField name="logo" type="string">
  The logo url of your store
</ResponseField>

<ResponseField name="primary_color" type="string">
  The hexadecimal primary color of your store
</ResponseField>
