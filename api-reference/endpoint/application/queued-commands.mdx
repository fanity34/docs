---
title: "Queued Commands"
api: "GET https://api.centralcart.com.br/api/application/queued_commands"
authMethod: "bearer"
description: "Get commands waiting to be executed."
---

### Response

<ResponseField name="id" type="number">
  The command ID
</ResponseField>

<ResponseField name="store_id" type="string">
  The store ID
</ResponseField>

<ResponseField name="command" type="string">
  The command string to be handled and executed
</ResponseField>

<ResponseField name="offline_execute" type="boolean">
  If true will be executed even with player offline
</ResponseField>

<ResponseField name="order" type="Order Object">
  <Expandable title="properties" defaultOpen={true}>
    <ResponseField name="client_identifier" type="string">
      The client identifier provided on checkout
    </ResponseField>
    <ResponseField name="client_email" type="string">
      The client email provided on checkout
    </ResponseField>
    <ResponseField name="client_name" type="string">
      The client name provided on checkout
    </ResponseField>
    <ResponseField name="client_discord" type="string | null">
      The client discord provided on checkout.
    </ResponseField>
    <ResponseField name="internal_id" type="string">
      The internal id of the order used to get detailed information if needed
    </ResponseField>
  </Expandable>
</ResponseField>
